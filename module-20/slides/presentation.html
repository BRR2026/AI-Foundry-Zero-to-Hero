from azure.ai.evaluation import AgentEvaluator
from azure.identity import DefaultAzureCredential

evaluator = AgentEvaluator(

        "subscription_id": "<sub-id>",
        "resource_group_name": "<rg>",
        "project_name": "<project>",
    },
    credential=DefaultAzureCredential()

results = evaluator.evaluate(

App Insights Integration

4-Step Setup

  1. Create an Application Insights resource
  2. Link to your Azure AI Foundry project
  3. Enable tracing in your agent code
  4. Verify traces in the portal
from azure.monitor.opentelemetry import configure_azure_monitor
from azure.ai.projects import AIProjectClient

# Step 1: Enable telemetry export
configure_azure_monitor(
    connection_string=os.environ["APPLICATIONINSIGHTS_CONNECTION_STRING"]
)

# Step 2: Create client with telemetry
client = AIProjectClient(
    credential=DefaultAzureCredential(),
    endpoint=os.environ["PROJECT_ENDPOINT"],
)
client.telemetry.enable()

OpenTelemetry Semantic Conventions

Azure AI Foundry follows the OpenTelemetry GenAI conventions for consistent trace attributes.

Model Attributes

gen_ai.system = "az.ai.agents"
gen_ai.request.model = "GPT-5.1"
gen_ai.operation.name = "chat"

Token Attributes

gen_ai.usage.input_tokens  = 1250
gen_ai.usage.output_tokens = 340

Agent-Specific Attributes

gen_ai.agent.id     = "asst_abc123"
gen_ai.thread.id    = "thread_xyz789"
gen_ai.thread.run.id = "run_def456"

Custom Span Enrichment

Add business context to traces for meaningful monitoring:

from opentelemetry import trace

tracer = trace.get_tracer("my-agent-app")

with tracer.start_as_current_span("process_request") as span:
    span.set_attribute("customer.id", customer_id)
    span.set_attribute("request.type", "refund_inquiry")
    span.set_attribute("request.priority", "high")

    response = agent_client.run(
        agent_id=agent_id, thread_id=thread_id
    )

    span.set_attribute("agent.task_completed", True)
    span.set_attribute("agent.total_tokens",
                     response.usage.total_tokens)

Debugging Workflow

  1. Identify the failed run — query App Insights for failed runs
  2. Inspect the full trace — view the distributed trace end-to-end
  3. Analyze tool call sequences — verify correct tool selection
  4. Review token usage — compare against expected ranges
  5. Test the fix — replay the conversation to verify
-- Find failed agent runs in the last 24 hours
dependencies
| where timestamp > ago(24h)
| where type == "az.ai.agents"
| where success == false
| project timestamp, id, name, duration, resultCode
| order by timestamp desc

KQL Dashboard Queries

Success Rate

dependencies
| where timestamp > ago(7d)
| where type == "az.ai.agents"
| summarize
    total = count(),
    ok = countif(success)
    by bin(timestamp, 1h)
| extend rate = round(
    100.0 * ok / total, 2)
| render timechart

Token Usage

dependencies
| where timestamp > ago(7d)
| where type == "az.ai.agents"
| extend tokens = toint(
    customDimensions[
      "gen_ai.usage.input_tokens"
    ])
| summarize
    avg_t = avg(tokens),
    p95 = percentile(tokens, 95)
    by bin(timestamp, 1d)

🛠️ Mini Hack: End-to-End Tracing

Set up complete observability for your agent:

  1. Configure azure-monitor-opentelemetry export
  2. Enable client.telemetry.enable() on AIProjectClient
  3. Add custom span attributes for business context
  4. Run 5 test conversations through your agent
  5. Build a KQL dashboard: success rates, latency, tokens
  6. Stretch: Set up an alert rule for success rate < 90%

⏱ Time: 15 minutes · 📋 Deliverable: App Insights dashboard with live agent traces

Best Practices

Evaluation

  • Automate in CI/CD pipelines
  • Use production data for test cases
  • Track metrics over time
  • Combine AI + human review

Debugging

  • Reproduce before fixing
  • Check tool outputs first
  • Version agent instructions

Tracing

  • Always enable in production
  • Use sampling (10–25%) for volume
  • Add business context to spans
  • Set up alerts from day 1
  • Implement log correlation

💰 Cost Tip: Use OTEL_TRACES_SAMPLER_ARG=0.1 to sample 10% in production.
